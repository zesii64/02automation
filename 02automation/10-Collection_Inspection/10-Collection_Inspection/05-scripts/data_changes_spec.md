### JavaScript Changes

**Rewrite `loadTrendChart()`** (old: lines 707-729):

```javascript
function loadTrendChart() {
    const grid = document.getElementById('module-charts-grid');
    grid.innerHTML = '';

    // Dispose old chart instances (if any)
    if (window.moduleCharts) {
        window.moduleCharts.forEach(c => c.dispose());
    }
    window.moduleCharts = [];

    MOCK_DATA.modules.forEach(module => {
        const isAtRisk = MOCK_DATA.riskModules.includes(module);
        const badgeClass = isAtRisk ? 'status-danger' : 'status-success';
        const badgeText = isAtRisk ? 'At Risk' : 'On Track';

        // Create card
        const card = document.createElement('div');
        card.className = 'module-trend-card';
        card.innerHTML = `
            <div class="module-card-header">
                <h4 class="module-card-title">Module ${module}</h4>
                <span class="status-badge ${badgeClass}">${badgeText}</span>
            </div>
            <div id="trend-chart-${module}" class="module-trend-chart"></div>
        `;
        grid.appendChild(card);

        // Initialize ECharts for this module
        const chartDom = document.getElementById(`trend-chart-${module}`);
        const chart = echarts.init(chartDom);
        window.moduleCharts.push(chart);

        const trendData = MOCK_DATA.moduleDailyTrends[module];
        const actualValues = trendData ? trendData.daily.map(d => d.actual) : [];
        const targetValue = trendData ? trendData.target : 0;

        chart.setOption({
            tooltip: { trigger: 'axis' },
            legend: { data: ['Actual', 'Target'], bottom: 0, itemGap: 20 },
            grid: { left: '3%', right: '4%', bottom: '15%', top: '10%', containLabel: true },
            xAxis: { type: 'category', data: generateDateLabels(30) },
            yAxis: {
                type: 'value',
                axisLabel: { formatter: (v) => (v / 1000) + 'k' }
            },
            series: [
                {
                    name: 'Actual',
                    type: 'line',
                    smooth: true,
                    data: actualValues,
                    lineStyle: { color: '#1e3a5f', width: 2 },
                    itemStyle: { color: '#1e3a5f' },
                    areaStyle: { color: 'rgba(30, 58, 95, 0.05)' }
                },
                {
                    name: 'Target',
                    type: 'line',
                    data: Array(30).fill(targetValue),
                    lineStyle: { color: '#059669', type: 'dashed', width: 2 },
                    itemStyle: { color: '#059669' },
                    symbol: 'none'
                }
            ]
        });
    });
}
```

**Chart ID naming**: `trend-chart-S0`, `trend-chart-S1`, `trend-chart-M1`, etc.

## Change 4: `moduleDailyTrends` Data Wiring

### Current (Wrong) Function

`generateTrendData(days)` (lines 731-737):
```javascript
function generateTrendData(days) {
    const data = [];
    for (let i = 0; i < days; i++) {
        data.push((70 + Math.random() * 25).toFixed(1));  // random 70-100%
    }
    return data;
}
```

**Problem**: Returns random percentages, not PHP amounts. No per-module targeting.

**Add new helper function** (before MOCK_DATA or alongside other generators):

```javascript
function generateModuleDailyData(days, min, max) {
    const data = [];
    for (let i = 0; i < days; i++) {
        const date = new Date();
        date.setDate(date.getDate() - days + i + 1);
        data.push({
            date: date.toISOString().split('T')[0],
            actual: Math.floor(min + Math.random() * (max - min))
        });
    }
    return data;
}
```

Note: `generateTrendData()` function can be **removed** or kept (unused) after this change.
